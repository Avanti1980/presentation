import numpy as np

def restore_lcs(b, x, i, j, lcs):  # LCS
    if i == -1 or j == -1:
        return
    if b[i, j] == -1:
        restore_lcs(b, x, i-1, j-1, lcs)
        lcs.append(x[i])
    elif b[i, j] == -2:
        restore_lcs(b, x, i-1, j, lcs)
    else:
        restore_lcs(b, x, i, j-1, lcs)
